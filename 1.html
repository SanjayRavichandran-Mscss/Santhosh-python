def binary_search(arr, x):
    start = 0
    end = len(arr) - 1
    while start <= end:
        mid = (start + end) // 2  
        if arr[mid] == x:
            return mid
        elif arr[mid] < x:
            start = mid + 1
        else:
            end = mid - 1
    return -1  
arr = [1, 3, 5, 7, 9, 11, 13, 15]
x = int(input("Enter the value: "))
index = binary_search(arr, x)
if index != -1:
    print(f"Number {x} found at index {index}")
else:
    print(f"Number {x} not found")
